🧱 Step 3 — Map Component (FULL)
import Map, {
  Source,
  Layer,
  Marker,
  Popup
} from "react-map-gl";
import "mapbox-gl/dist/mapbox-gl.css";
import { useState } from "react";
import { stores, paths } from "./mallData";

export default function MallMap() {
  const [selected, setSelected] = useState(null);
  const [level, setLevel] = useState("UL");
  const [filter, setFilter] = useState("all");

  const filteredStores = {
    ...stores,
    features: stores.features.filter(
      f =>
        f.properties.level === level &&
        (filter === "all" || f.properties.category === filter)
    )
  };

  return (
    <div className="relative w-full h-screen">
      <Map
        mapboxAccessToken={process.env.REACT_APP_MAPBOX_TOKEN}
        initialViewState={{
          longitude: -75.384,
          latitude: 40.089,
          zoom: 17
        }}
        style={{ width: "100%", height: "100%" }}
        mapStyle="mapbox://styles/mapbox/light-v11"
      >
        {/* PATHS */}
        <Source id="paths" type="geojson" data={paths}>
          <Layer
            id="paths-layer"
            type="line"
            paint={{
              "line-color": "#ff007a",
              "line-width": 3
            }}
          />
        </Source>

        {/* STORES */}
        <Source id="stores" type="geojson" data={filteredStores}>
          <Layer
            id="store-layer"
            type="circle"
            paint={{
              "circle-radius": 6,
              "circle-color": "#111"
            }}
          />
        </Source>

        {/* CLICK HANDLER */}
        {filteredStores.features.map((store) => (
          <Marker
            key={store.properties.id}
            longitude={store.geometry.coordinates[0]}
            latitude={store.geometry.coordinates[1]}
            onClick={() => setSelected(store)}
          >
            <div className="w-3 h-3 bg-pink-500 rounded-full cursor-pointer" />
          </Marker>
        ))}

        {/* POPUP */}
        {selected && (
          <Popup
            longitude={selected.geometry.coordinates[0]}
            latitude={selected.geometry.coordinates[1]}
            onClose={() => setSelected(null)}
          >
            <h3 className="font-bold">{selected.properties.name}</h3>
            <p>{selected.properties.category}</p>
          </Popup>
        )}
      </Map>

      {/* SIDEBAR */}
      <div className="absolute top-6 left-6 w-80 bg-white rounded-2xl shadow-xl p-4">
        <input
          placeholder="Search..."
          className="w-full p-2 rounded-full bg-gray-100"
        />

        <div className="flex gap-2 mt-3">
          <button onClick={() => setFilter("all")}>All</button>
          <button onClick={() => setFilter("shopping")}>Shopping</button>
          <button onClick={() => setFilter("dining")}>Dining</button>
        </div>

        <div className="flex gap-2 mt-3">
          <button onClick={() => setLevel("UL")}>UL</button>
          <button onClick={() => setLevel("ML")}>ML</button>
          <button onClick={() => setLevel("LL")}>LL</button>
        </div>
      </div>
    </div>
  );
}

import Map, { Source, Layer } from "react-map-gl";
import { useState } from "react";
import mallGeo from "./kop.geojson";

export default function KOPMap() {
  const [level, setLevel] = useState("UL");
  const [category, setCategory] = useState("all");

  const filtered = {
    ...mallGeo,
    features: mallGeo.features.filter(
      f =>
        f.properties.level === level &&
        (category === "all" ||
          f.properties.category === category)
    )
  };

  return (
    <div className="h-screen w-full relative">
      <Map
        mapboxAccessToken={process.env.REACT_APP_MAPBOX}
        initialViewState={{
          longitude: -75.3886,
          latitude: 40.0888,
          zoom: 18
        }}
        mapStyle="mapbox://styles/mapbox/light-v11"
      >
        <Source type="geojson" data={filtered}>
          <Layer
            id="stores"
            type="fill"
            paint={{
              "fill-color": "#111",
              "fill-opacity": 0.6
            }}
          />
        </Source>
      </Map>

      {/* UI PANEL */}
      <div className="absolute left-6 top-6 bg-white p-4 rounded-xl w-80 shadow">
        <input
          placeholder="Search stores..."
          className="w-full p-2 bg-gray-100 rounded"
        />

        <div className="flex gap-2 mt-3">
          <button onClick={() => setCategory("all")}>All</button>
          <button onClick={() => setCategory("shopping")}>Shopping</button>
          <button onClick={() => setCategory("dining")}>Dining</button>
        </div>

        <div className="flex gap-2 mt-3">
          <button onClick={() => setLevel("UL")}>UL</button>
          <button onClick={() => setLevel("ML")}>ML</button>
          <button onClick={() => setLevel("LL")}>LL</button>
        </div>
      </div>
    </div>
  );
}
